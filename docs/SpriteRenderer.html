SpriteRenderer

Renders a Sprite for 2D graphics.

Usage

public class Main extends Scene {

    GameObject mario = new GameObject(new Transform(600, 200, 50, 50), 2);
    
    public void awake() {	
        setDefaultBackground(Color.BLACK);
        camera = new Camera();
        
        mario.addComponent(new SpriteRenderer(new Sprite(Assets.getTexture("src/assets/images/marioSprite.png"))));
    }

    public void update() {

    }
}
